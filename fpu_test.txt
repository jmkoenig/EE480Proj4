cii $r1, 0x1
cup $r1, 0x0 ; r1=1
cii $r2, 0x0
cup $r2, 0x3f ; AKA r2=0.5 in decimal
cii $r4, 0x01 ; r4=0x0101

i2f $r1
f2i $r1 ; r1=1
dup $r2, $r1 ; r2=1
negi $r2 ; r2=-1
addi $r1, $r2 ; r1=0
bz $r1, arithf
trap

arith:
cii $r1, 0x80
cup $r1, 0x3f ; r1=0x3f80 (1 in float)
cii $r2, 0x80
cup $r2, 0xbf ; r2=0xbf80(-1)
addf $r1, $r2 ; r1=0
bz $r1, mul
trap

mul:
cii $r2, 0x0
cup $r2, 0x40 ; r2=0x4000(2)
mulf $r2, $r1 ; r2=0
bz $r2, inv
trap

inv:
cii $r3, 0x40 ; r3=0x4040(3)
invf $r3 ; r3=1/3
cii $r2, 0x40
cup $r2, 0xc0 ; r2=0xc040(-1/3)
addf $r3, $r2 ; r3=0
bz $r3, neg
trap

neg:
dup $r3, $r2 ; r3=-1/3
negf $r3 ; r3=1/3
addf $r3, $r2 ; r3=0
bz $r3, posits
trap

posits:
ii2pp $r4
invpp $r4
invpp $r4
pp2ii $r4 ; r4=0x0101
dup $r5, $r4
negii $r5
addii $r5, $r4 ; r5=0
bz $r5, fp
trap

fp:
f2pp $r2 ; (r2 still -1/3)
pp2f $r2
dup $r3, $r2 ; r3=-1/3
negf $r3 ; r3=1/3
addf $r3, $r2 ; r3=0
bz $r3, pps
trap

pps:
; if ii2pp/pp2ii works:
ii2pp $r4 ; (r4 still 0x0101)
dup $r5, $r4
addpp $r5, $r4
pp2ii $r5 ; r5=0x0202
dup $r4, $r5 ; r4=0x0202
negii $r4 ; r4=-0x8282
addii $r4, $r5
bz $r4, ppmul
trap

ppmul:
ii2pp $r5 ; (r5 still 0x0202)
mulpp $r5, $r5 ; r5=0x0404
pp2ii $r5
dup $r4, $r5 ; r4=0x0404
negii $r4 ; r4=-0x8484
addii $r4, $r5
bz $r4, done
trap

done:
trap
